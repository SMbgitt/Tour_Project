create table Continent( 
Id int primary key, 
Name varchar(30) 
); 
 
 
 
create table Country( 
Id int primary key, 
ContinentId int, 
Name varchar(30), 
Description varchar(2000), 
Timezone varchar(2000), 
Weather varchar(2000), 
Kitchen varchar(2000), 
foreign key (ContinentId)references Continent(Id) 
); 
 
 
create table Fact( 
Id int primary key, 
CountryId int, 
Description varchar(2000), 
foreign key (CountryId)references Country(Id) 
); 
 
create table Currency( 
Id int primary key, 
Description varchar(2000)
); 
 
create table CountryCurrency( 
Id int primary key, 
CountryId int, 
CurrencyId int, 
foreign key (CountryId)references Country(Id), 
foreign key (CurrencyId)references Currency(Id) 
); 
 
create table Danger( 
Id int primary key,
Description varchar(2000) 
); 
 
 
 
create table CountryDangerous( 
Id int primary key, 
CountryId int, 
DangerId int, 
foreign key(CountryId) references Country(Id), 
foreign key(DangerId) references Danger(Id), 
); 
 
 
 
create table Sight( 
Id int primary key, 
CountryId int, 
Description varchar(2000), 
foreign key(CountryId) references Country(Id) 
); 
 
 
 
create table Picture( 
Id int primary key, 
CountryId int, 
Link varchar(100), 
foreign key(CountryId) references Country(Id) 
);
